//Показать цену, количество на складе и общее количЕство заказанных единиц для 
//каждого наименования товара, если для него общее количество заказанных единиц 
//превышает 75 процентов от колиЧЕства товара на складе.
SELECT DESCRIPTION, PRICE, QTY_ON_HAND, SUM(QTY)
FROM PRODUCTS, ORDERS
WHERE MFR = MFR ID
AND PRODUCT = PRODUCT_ID
GROUP ВУ MFR_ID, PRODUCT_ID, DESCRIPTION, PRICE, QTY_ON_HAND
НAVING SUМ(QTY) > (.75 * QTY_ON_HAND)
ORDER ВУ QTY_ON_НAND DESC;
